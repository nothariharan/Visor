#: Advanced Organization & Power User Features

## Overview

We'll implement several powerful features to give users control over their codebase visualization:

1. **"Organize" Button** - Smart filtering to show only critical execution paths
2. **Custom Grouping Boxes** - Manually organize nodes into visual containers
3. **Additional Power Features** - Layers, bookmarks, annotations, etc.

---

## Feature 1: Smart Organization - "Show Only Critical Path"

### Concept

When user clicks "Organize", hide unnecessary files and show only:
- **Entry points** (index.js, main.tsx, App.jsx)
- **Files in the execution chain** (files that entry points import, recursively)
- **Core utilities** (files imported by multiple modules)

**Before Organize:**
```
150 files visible, messy hairball
```

**After Organize:**
```
20 critical files, clean dependency tree
Entry → App → Routes → Components → Utilities
```

### Implementation

#### Step 1: Backend - Critical Path Algorithm

```javascript
// server/analyzer/CriticalPathAnalyzer.js

export class CriticalPathAnalyzer {
  constructor(graph) {
    this.nodes = graph.nodes;
    this.edges = graph.edges;
    this.adjacencyList = this.buildAdjacencyList();
  }
  
  buildAdjacencyList() {
    const adj = {};
    
    this.nodes.forEach(node => {
      adj[node.id] = { imports: [], importedBy: [] };
    });
    
    this.edges.forEach(edge => {
      adj[edge.source].imports.push(edge.target);
      adj[edge.target].importedBy.push(edge.source);
    });
    
    return adj;
  }
  
  /**
   * Identify critical files based on multiple criteria
   */
  findCriticalPath() {
    const critical = new Set();
    
    // 1. Find entry points
    const entryPoints = this.findEntryPoints();
    entryPoints.forEach(id => critical.add(id));
    
    // 2. Traverse dependency tree from entry points (BFS)
    const visited = new Set();
    const queue = [...entryPoints];
    
    while (queue.length > 0) {
      const nodeId = queue.shift();
      
      if (visited.has(nodeId)) continue;
      visited.add(nodeId);
      critical.add(nodeId);
      
      // Add all dependencies
      const deps = this.adjacencyList[nodeId]?.imports || [];
      deps.forEach(dep => {
        // Skip test files, stories, mocks
        if (this.isSupplementaryFile(dep)) return;
        
        queue.push(dep);
      });
    }
    
    // 3. Add high-centrality nodes (imported by many files)
    const centralNodes = this.findCentralNodes(5);  // Top 5 most-imported
    centralNodes.forEach(id => critical.add(id));
    
    return {
      critical: Array.from(critical),
      entryPoints,
      centralNodes
    };
  }
  
  /**
   * Find entry points (files with no importers or specific names)
   */
  findEntryPoints() {
    const entryPoints = [];
    
    // Common entry point patterns
    const entryPatterns = [
      /index\.(js|jsx|ts|tsx)$/,
      /main\.(js|jsx|ts|tsx)$/,
      /App\.(js|jsx|ts|tsx)$/,
      /app\.(js|jsx|ts|tsx)$/,
      /_app\.(js|jsx|ts|tsx)$/,  // Next.js
      /_document\.(js|jsx|ts|tsx)$/  // Next.js
    ];
    
    this.nodes.forEach(node => {
      const isNamedEntry = entryPatterns.some(pattern => pattern.test(node.id));
      const hasNoImporters = (this.adjacencyList[node.id]?.importedBy.length || 0) === 0;
      
      if (isNamedEntry || hasNoImporters) {
        entryPoints.push(node.id);
      }
    });
    
    return entryPoints;
  }
  
  /**
   * Find nodes with highest import count (centrality)
   */
  findCentralNodes(topN = 5) {
    const centrality = this.nodes.map(node => ({
      id: node.id,
      score: this.adjacencyList[node.id]?.importedBy.length || 0
    }));
    
    return centrality
      .sort((a, b) => b.score - a.score)
      .slice(0, topN)
      .map(n => n.id);
  }
  
  /**
   * Check if file is supplementary (test, story, mock, etc.)
   */
  isSupplementaryFile(filePath) {
    const supplementaryPatterns = [
      /\.test\.(js|jsx|ts|tsx)$/,
      /\.spec\.(js|jsx|ts|tsx)$/,
      /\.stories\.(js|jsx|ts|tsx)$/,
      /\.mock\.(js|jsx|ts|tsx)$/,
      /__tests__\//,
      /__mocks__\//,
      /\.d\.ts$/  // TypeScript declarations
    ];
    
    return supplementaryPatterns.some(pattern => pattern.test(filePath));
  }
  
  /**
   * Apply critical path filter to graph
   */
  applyFilter(mode = 'critical') {
    const { critical, entryPoints, centralNodes } = this.findCriticalPath();
    
    const filteredNodes = this.nodes.map(node => ({
      ...node,
      hidden: !critical.includes(node.id),
      data: {
        ...node.data,
        isEntryPoint: entryPoints.includes(node.id),
        isCentral: centralNodes.includes(node.id),
        criticalReason: this.getCriticalReason(node.id, entryPoints, centralNodes)
      }
    }));
    
    const filteredEdges = this.edges.filter(edge => 
      critical.includes(edge.source) && critical.includes(edge.target)
    );
    
    return {
      nodes: filteredNodes,
      edges: filteredEdges,
      stats: {
        total: this.nodes.length,
        critical: critical.length,
        hidden: this.nodes.length - critical.length
      }
    };
  }
  
  getCriticalReason(nodeId, entryPoints, centralNodes) {
    if (entryPoints.includes(nodeId)) return 'Entry Point';
    if (centralNodes.includes(nodeId)) return 'High Usage';
    return 'In Execution Path';
  }
}
```

#### Step 2: Backend API Endpoint

```javascript
// server/index.js

app.post('/api/graph/organize', express.json(), async (req, res) => {
  const { mode } = req.body;  // 'critical', 'all', 'custom'
  
  try {
    const graph = watcher.getLatestGraph();
    const analyzer = new CriticalPathAnalyzer(graph);
    
    let result;
    
    switch (mode) {
      case 'critical':
        result = analyzer.applyFilter('critical');
        break;
      
      case 'all':
        result = graph;  // Show everything
        break;
      
      default:
        result = graph;
    }
    
    res.json(result);
    
  } catch (error) {
    console.error('Error organizing graph:', error);
    res.status(500).json({ error: error.message });
  }
});
```

#### Step 1: Backend - Save Group Configurations

```javascript
// server/groups/GroupManager.js

import fs from 'fs/promises';
import path from 'path';

export class GroupManager {
  constructor(projectRoot) {
    this.projectRoot = projectRoot;
    this.groupsFile = path.join(projectRoot, '.visor', 'groups.json');
  }
  
  async loadGroups() {
    try {
      const content = await fs.readFile(this.groupsFile, 'utf-8');
      return JSON.parse(content);
    } catch {
      return [];
    }
  }
  
  async saveGroups(groups) {
    const visorDir = path.join(this.projectRoot, '.visor');
    
    // Create .visor directory if it doesn't exist
    try {
      await fs.mkdir(visorDir, { recursive: true });
    } catch {}
    
    await fs.writeFile(
      this.groupsFile,
      JSON.stringify(groups, null, 2),
      'utf-8'
    );
  }
  
  async createGroup(groupData) {
    const groups = await this.loadGroups();
    
    const newGroup = {
      id: `group-${Date.now()}`,
      name: groupData.name,
      color: groupData.color || '#3b82f6',
      nodes: groupData.nodes || [],
      position: groupData.position || { x: 0, y: 0 },
      size: groupData.size || { width: 400, height: 300 },
      collapsed: false,
      createdAt: new Date().toISOString()
    };
    
    groups.push(newGroup);
    await this.saveGroups(groups);
    
    return newGroup;
  }
  
  async updateGroup(groupId, updates) {
    const groups = await this.loadGroups();
    const index = groups.findIndex(g => g.id === groupId);
    
    if (index !== -1) {
      groups[index] = { ...groups[index], ...updates };
      await this.saveGroups(groups);
      return groups[index];
    }
    
    return null;
  }
  
  async deleteGroup(groupId) {
    const groups = await this.loadGroups();
    const filtered = groups.filter(g => g.id !== groupId);
    await this.saveGroups(filtered);
  }
  
  async addNodeToGroup(groupId, nodeId) {
    const groups = await this.loadGroups();
    const group = groups.find(g => g.id === groupId);
    
    if (group && !group.nodes.includes(nodeId)) {
      group.nodes.push(nodeId);
      await this.saveGroups(groups);
    }
  }
  
  async removeNodeFromGroup(groupId, nodeId) {
    const groups = await this.loadGroups();
    const group = groups.find(g => g.id === groupId);
    
    if (group) {
      group.nodes = group.nodes.filter(n => n !== nodeId);
      await this.saveGroups(groups);
    }
  }
}
```

```javascript
// src/hooks/useNodeGrouping.js

import { useCallback } from 'react';
import { useGraphStore } from '../stores/graphStore';

export function useNodeGrouping() {
  const groups = useGraphStore(state => state.groups);
  const addNodeToGroup = useGraphStore(state => state.addNodeToGroup);
  
  const handleNodeDragStop = useCallback((event, node) => {
    const nodeRect = event.target.getBoundingClientRect();
    
    // Check if node is inside any group box
    for (const group of groups) {
      const groupRect = document
        .querySelector(`[data-group-id="${group.id}"]`)
        ?.getBoundingClientRect();
      
      if (!groupRect) continue;
      
      const isInside = (
        nodeRect.left >= groupRect.left &&
        nodeRect.right <= groupRect.right &&
        nodeRect.top >= groupRect.top &&
        nodeRect.bottom <= groupRect.bottom
      );
      
      if (isInside) {
        addNodeToGroup(group.id, node.id);
        break;
      }
    }
  }, [groups, addNodeToGroup]);
  
  return { handleNodeDragStop };
}
```

```javascript
// Auto-assign logic
function autoAssignLayers(nodes) {
  return nodes.map(node => {
    // UI components
    if (/components?/i.test(node.id) || /\.jsx$/.test(node.id)) {
      return { ...node, layer: 'presentation' };
    }
    
    // Services/API
    if (/services?|api/i.test(node.id)) {
      return { ...node, layer: 'business' };
    }
    
    // Models/Database
    if (/models?|database|db/i.test(node.id)) {
      return { ...node, layer: 'data' };
    }
    
    return node;
  });
}
```

### 3.5 Dependency Heat Map

**Concept:** Color-code nodes by complexity/coupling

```javascript
// src/utils/heatmap.js

export function calculateHeatMap(nodes, edges) {
  const adjacency = buildAdjacencyList(edges);
  
  return nodes.map(node => {
    const imports = adjacency[node.id]?.imports.length || 0;
    const importedBy = adjacency[node.id]?.importedBy.length || 0;
    
    // Coupling score
    const coupling = imports + importedBy;
    
    // Color gradient: Green → Yellow → Red
    let color;
    if (coupling < 5) color = '#10b981';       // Green (low)
    else if (coupling < 15) color = '#f59e0b'; // Yellow (medium)
    else color = '#ef4444';                    // Red (high)
    
    return {
      ...node,
      style: {
        ...node.style,
        background: color,
        opacity: 0.3 + (coupling / 30) * 0.7  // More coupled = more opaque
      }
    };
  });
}
```

```javascript
// src/components/TimeTravelControls.jsx

export function TimeTravelControls() {
  const [commits, setCommits] = useState([]);
  const [currentCommit, setCurrentCommit] = useState(null);
  
  useEffect(() => {
    fetch('/api/git/history')
      .then(res => res.json())
      .then(setCommits);
  }, []);
  
  const jumpToCommit = async (commitHash) => {
    const response = await fetch(`/api/graph/at-commit/${commitHash}`);
    const graph = await response.json();
    
    useGraphStore.getState().setGraph(graph);
    setCurrentCommit(commitHash);
  };
  
  return (
    <div className="time-travel">
      <h4>📅 History</h4>
      <input
        type="range"
        min={0}
        max={commits.length - 1}
        onChange={(e) => jumpToCommit(commits[e.target.value].hash)}
      />
      
      {currentCommit && (
        <div className="commit-info">
          Viewing: {commits.find(c => c.hash === currentCommit)?.message}
        </div>
      )}
    </div>
  );
}
```

```javascript
export function AIInsights() {
  const [insights, setInsights] = useState([]);

        <div key={i} className={`insight ${insight.severity}`}>
          <h5>{insight.title}</h5>
          <p>{insight.description}</p>
          
          {insight.action && (
            <button onClick={() => applyFix(insight.action)}>
              Apply Fix
            </button>
          )}
        </div>
      ))}
    </div>
  );
}

// Example backend AI analysis
async function analyzeWithAI(graph) {
  const insights = [];
  
  // Detect circular dependencies
  const circular = detectCircular(graph.edges);
  if (circular.length > 0) {
    insights.push({
      severity: 'high',
      title: 'Circular Dependencies Detected',
      description: `Found ${circular.length} circular dependency chains`,
      action: { type: 'highlight', nodes: circular }
    });
  }
  
  // Detect god objects (files with too many connections)
  const godObjects = graph.nodes.filter(n => 
    n.data.imports + n.data.importedBy > 20
  );
  
  if (godObjects.length > 0) {
    insights.push({
      severity: 'medium',
      title: 'High Coupling Detected',
      description: `${godObjects.length} files have excessive dependencies`,
      action: { type: 'suggest-split', nodes: godObjects }
    });
  }
  
  return { insights };
}
```
